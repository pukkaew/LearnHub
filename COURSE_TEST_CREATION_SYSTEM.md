// ELIGIBILITY CHECKS
function canTakeTest(user, test, course) {
    // 1. Check enrollment
    if (!isEnrolled(user, course)) {
        return { allowed: false, reason: 'Not enrolled in course' };
    }

    // 2. Check prerequisites
    const prereqs = getPrerequisites(course);
    for (const prereq of prereqs) {
        if (!hasCompleted(user, prereq)) {
            return { allowed: false, reason: `Must complete ${prereq.name} first` };
        }
    }

    // 3. Check lesson completion (if required)
    if (test.required_for_completion) {
        const progress = getCourseProgress(user, course);
        if (progress < 100) {
            return { allowed: false, reason: 'Complete all lessons first' };
        }
    }

    // 4. Check attempt limit
    const attempts = getAttempts(user, test);
    if (test.attempts_allowed && attempts.length >= test.attempts_allowed) {
        return { allowed: false, reason: 'Maximum attempts reached' };
    }

    // 5. Check if already passed
    if (attempts.some(a => a.passed)) {
        return { allowed: true, warning: 'Already passed. Taking again will replace score.' };
    }

    // 6. Check availability dates
    const now = new Date();
    if (test.available_from && now < test.available_from) {
        return { allowed: false, reason: `Available from ${test.available_from}` };
    }
    if (test.available_until && now > test.available_until) {
        return { allowed: false, reason: 'Test deadline passed' };
    }

    return { allowed: true };
}

// SUBMISSION VALIDATION
function validateSubmission(attempt, test) {
    // 1. Check time limit
    if (test.time_limit) {
        const elapsed = (now - attempt.started_at) / 1000 / 60;  // minutes
        if (elapsed > test.time_limit) {
            return { valid: false, reason: 'Time limit exceeded' };
        }
    }

    // 2. Check all questions answered (if required)
    const answered = getAnsweredCount(attempt);
    if (test.require_all_answers && answered < test.total_questions) {
        return { valid: false, reason: 'Answer all questions' };
    }

    // 3. Check not already submitted
    if (attempt.status === 'submitted') {
        return { valid: false, reason: 'Already submitted' };
    }

    return { valid: true };
}

// AUTO-GRADING (Multiple Choice, True/False)
function autoGrade(attempt, test) {
    let totalScore = 0;
    let maxScore = 0;

    for (const question of test.questions) {
        maxScore += question.points;

        const userAnswer = getUserAnswer(attempt, question);
        if (!userAnswer) continue;

        if (question.type === 'multiple-choice') {
            const correctOptions = question.options.filter(o => o.is_correct);
            const selectedOptions = userAnswer.selected_option_ids;

            // All correct selected AND no incorrect selected
            const allCorrect = correctOptions.every(o => selectedOptions.includes(o.option_id));
            const noneIncorrect = selectedOptions.every(id =>
                correctOptions.some(o => o.option_id === id)
            );

            if (allCorrect && noneIncorrect) {
                totalScore += question.points;
                userAnswer.is_correct = true;
                userAnswer.points_earned = question.points;
            } else {
                userAnswer.is_correct = false;
                userAnswer.points_earned = 0;
            }
        }
    }

    const percentage = (totalScore / maxScore) * 100;
    const passed = percentage >= test.passing_marks;

    return {
        score: totalScore,
        max_score: maxScore,
        percentage: percentage,
        passed: passed
    };
}

// MANUAL GRADING (Essay, Short Answer)
function requiresManualGrading(test) {
    return test.questions.some(q =>
        ['essay', 'short_answer'].includes(q.type) && !q.has_auto_gradeable_answers
    );
}

// CERTIFICATE ISSUANCE
function shouldIssueCertificate(user, course, testResult) {
    // 1. Course allows certificates
    if (!course.allow_certificate) {
        return false;
    }

    // 2. Test passed (if test required)
    if (course.test_id) {
        if (!testResult || !testResult.passed) {
            return false;
        }
    }

    // 3. Course completed
    const progress = getCourseProgress(user, course);
    if (progress < 100) {
        return false;
    }

    // 4. Not already issued (or expired)
    const existing = getCertificate(user, course);
    if (existing && !existing.is_revoked) {
        if (!existing.valid_until || existing.valid_until > new Date()) {
            return false;  // Still valid
        }
    }

    return true;
}

// CERTIFICATE VALIDITY
function calculateValidity(course) {
    if (!course.certificate_validity) {
        return null;  // Unlimited
    }

    const issued = new Date();
    const valid_until = new Date(issued);
    valid_until.setDate(valid_until.getDate() + parseInt(course.certificate_validity));

    return valid_until;
}

// CERTIFICATE CODE FORMAT
function generateCertificateCode(certificate_id, year) {
    return `CERT-${year}-${String(certificate_id).padStart(7, '0')}`;
    // Example: CERT-2025-0000050
}
